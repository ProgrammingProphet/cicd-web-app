pipeline details (simplified) 

name: CI/CD
on: push
jobs:
  build-and-deploy:
    runs-on: ubuntu-latest
    steps:
      - checkout
      - docker build -t myapp:${{ github.sha }}
      - docker push myapp:${{ github.sha }}
      - ssh vps "docker pull myapp:${{ github.sha }} && docker run -d ..."
